# 在项目中使用编译好的 Qt

本文档介绍如何在你的项目中使用已编译好的 Qt 库。

## 方法一：直接设置 CMAKE_PREFIX_PATH

这是最简单的方法，适合独立项目。

### 命令行指定

```bash
cmake -DCMAKE_PREFIX_PATH=/opt/qt-6.8.1 ..
cmake --build .
```

### CMakeLists.txt 中指定

```cmake
cmake_minimum_required(VERSION 3.16)
project(MyApp)

# 设置 Qt 前缀路径
set(CMAKE_PREFIX_PATH "/opt/qt-6.8.1")

# 查找 Qt 模块
find_package(Qt6 REQUIRED COMPONENTS
    Core
    Gui
    Widgets
    Network
    Quick
    Qml
)

# 创建可执行文件
add_executable(myapp
    src/main.cpp
    src/mainwindow.cpp
    src/mainwindow.h
)

# 链接 Qt 库
target_link_libraries(myapp PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::Network
    Qt6::Quick
    Qt6::Qml
)
```

### 使用环境变量

在 `~/.bashrc` 或项目脚本中设置：

```bash
export CMAKE_PREFIX_PATH=/opt/qt-6.8.1
```

这样 CMake 会自动查找 Qt。

## 方法二：作为 third-party 引入到其他仓库

适合大型项目或需要版本控制的情况。

### 目录结构

```
my-project/
├── CMakeLists.txt
├── src/
├── third-party/
│   └── qt/
│       └── qt-6.8.1/
│           ├── bin/
│           ├── lib/
│           ├── include/
│           ├── cmake/
│           └── ...
└── ...
```

### CMakeLists.txt 配置

```cmake
cmake_minimum_required(VERSION 3.16)
project(MyApp)

# 设置第三方库根目录
set(THIRD_PARTY_DIR "${CMAKE_SOURCE_DIR}/third-party")

# 设置 Qt 路径
set(QT_ROOT "${THIRD_PARTY_DIR}/qt/qt-6.8.1")

# 将 Qt 添加到 CMAKE_PREFIX_PATH
list(APPEND CMAKE_PREFIX_PATH "${QT_ROOT}")

# 查找 Qt
find_package(Qt6 REQUIRED COMPONENTS
    Core
    Gui
    Widgets
)

# 创建可执行文件
add_executable(myapp
    src/main.cpp
)

# 链接 Qt
target_link_libraries(myapp PRIVATE Qt6::Core Qt6::Gui Qt6::Widgets)

# 设置运行时路径（RPATH）
if(NOT CMAKE_INSTALL_RPATH)
    set(CMAKE_INSTALL_RPATH "${QT_ROOT}/lib")
endif()
```

### 使用 Git Submodule（推荐）

如果你将 Qt 代码或配置放在 Git 仓库中：

```bash
# 添加 submodule
git submodule add https://github.com/your-qt-config.git third-party/qt

# 更新 submodule
git submodule update --init --recursive
```

### 使用 Git Sparse Checkout（节省空间）

如果 Qt 在远程仓库：

```bash
# 启用 sparse checkout
mkdir -p third-party/qt
cd third-party/qt
git init
git sparse-checkout init
git sparse-checkout set bin lib include cmake plugins
git remote add origin https://github.com/your-repo/qt-6.8.1.git
git pull origin main
```

## 方法三：使用 Qt Creator 配置

### 配置 Qt 版本

1. 打开 Qt Creator
2. 进入 **Edit** -> **Preferences** -> **Kits**
3. 在 **Qt Versions** 标签页添加新版本：
   - **Name**: Qt 6.8.1 Custom
   - **qmake path**: `/opt/qt-6.8.1/bin/qmake`

### 配置 CMake 工具

1. 在 **Preferences** -> **Kits** -> **CMake**
2. 添加或确认 CMake 路径

### 配置 Kit

1. 在 **Preferences** -> **Kits** -> **Kits**
2. 添加新 Kit 或编辑现有 Kit：
   - **Name**: Desktop (Custom Qt)
   - **Device type**: Desktop
   - **Compiler**: 选择系统编译器
   - **CMake**: 选择 CMake 工具
   - **Qt version**: Qt 6.8.1 Custom

### 项目配置

在 `.pro` 文件中（使用 qmake）：

```qmake
# 指定 Qt 安装路径
QTDIR = /opt/qt-6.8.1

# 指定 qmake 路径（如果不在 PATH 中）
# QMAKE = $$QTDIR/bin/qmake

# 所需模块
QT += core gui widgets network

# 包含路径
INCLUDEPATH += $$QTDIR/include

# 库路径
LIBS += -L$$QTDIR/lib

# 目标
TARGET = myapp
SOURCES += src/main.cpp
```

在 Qt Creator 中使用 CMake 时，确保 Kit 配置正确。

## 示例 CMakeLists.txt

### 完整示例：Qt Widgets 应用

```cmake
cmake_minimum_required(VERSION 3.16)
project(MyQtApp VERSION 1.0.0 LANGUAGES CXX)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 设置 Qt 路径（可选，也可以通过环境变量设置）
set(QT_PATH "/opt/qt-6.8.1" CACHE PATH "Qt installation path")

# 添加 Qt 到 CMAKE_PREFIX_PATH
if(EXISTS "${QT_PATH}")
    list(APPEND CMAKE_PREFIX_PATH "${QT_PATH}")
endif()

# 启用自动 MOC、UIC 和 RCC
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# 查找 Qt
find_package(Qt6 REQUIRED COMPONENTS
    Core
    Gui
    Widgets
)

# 收集源文件
file(GLOB_RECURSE SOURCES
    src/*.cpp
    src/*.h
    src/*.ui
)

# 创建可执行文件
add_executable(${PROJECT_NAME}
    ${SOURCES}
)

# 链接 Qt 库
target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
)

# 包含目录
target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_SOURCE_DIR}/src
)

# 设置输出目录
set_target_properties(${PROJECT_NAME} PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# 安装规则
install(TARGETS ${PROJECT_NAME}
    RUNTIME DESTINATION bin
)
```

### 完整示例：Qt Quick 应用

```cmake
cmake_minimum_required(VERSION 3.16)
project(MyQuickApp VERSION 1.0.0 LANGUAGES CXX)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 设置 Qt 路径
set(QT_PATH "/opt/qt-6.8.1" CACHE PATH "Qt installation path")
if(EXISTS "${QT_PATH}")
    list(APPEND CMAKE_PREFIX_PATH "${QT_PATH}")
endif()

# 启用自动 MOC、UIC 和 RCC
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# 查找 Qt
find_package(Qt6 REQUIRED COMPONENTS
    Core
    Quick
    Qml
    Gui
)

# 收集源文件
file(GLOB_RECURSE SOURCES
    src/*.cpp
    src/*.h
)

# 收集 QML 资源
set(QML_SOURCES
    qml/main.qml
    qml/App.qml
)

# 创建可执行文件
qt_add_executable(${PROJECT_NAME}
    ${SOURCES}
)

# 链接 Qt 库
qt_add_qml_module(${PROJECT_NAME}
    URI MyApp
    VERSION 1.0
    QML_FILES ${QML_SOURCES}
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt6::Quick
    Qt6::Qml
    Qt6::Gui
)

# 设置输出目录
set_target_properties(${PROJECT_NAME} PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# 安装规则
install(TARGETS ${PROJECT_NAME}
    BUNDLE DESTINATION .
    RUNTIME DESTINATION bin
)

# 安装 QML 文件
install(DIRECTORY qml/
    DESTINATION qml
)
```

### 完整示例：库项目

```cmake
cmake_minimum_required(VERSION 3.16)
project(MyLib VERSION 1.0.0 LANGUAGES CXX)

# 设置 C++ 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 设置 Qt 路径
set(QT_PATH "/opt/qt-6.8.1" CACHE PATH "Qt installation path")
if(EXISTS "${QT_PATH}")
    list(APPEND CMAKE_PREFIX_PATH "${QT_PATH}")
endif()

# 启用自动 MOC
set(CMAKE_AUTOMOC ON)

# 查找 Qt
find_package(Qt6 REQUIRED COMPONENTS
    Core
    Network
)

# 创建库
add_library(${PROJECT_NAME} SHARED
    src/mylib.cpp
    src/mylib.h
)

target_link_libraries(${PROJECT_NAME} PUBLIC
    Qt6::Core
    Qt6::Network
)

# 导出头文件
target_include_directories(${PROJECT_NAME} PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)

# 生成导出头文件
generate_export_header(${PROJECT_NAME})

# 安装
install(TARGETS ${PROJECT_NAME}
    EXPORT MyLibTargets
    ARCHIVE DESTINATION lib
    LIBRARY DESTINATION lib
    RUNTIME DESTINATION bin
)

install(FILES
    include/mylib.h
    ${CMAKE_CURRENT_BINARY_DIR}/mylib_export.h
    DESTINATION include
)

# 导出目标
install(EXPORT MyLibTargets
    FILE MyLibTargets.cmake
    NAMESPACE MyLib::
    DESTINATION lib/cmake/MyLib
)

# 创建配置文件
include(CMakePackageConfigHelpers)
write_basic_package_version_file(
    "${CMAKE_CURRENT_BINARY_DIR}/MyLibConfigVersion.cmake"
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY AnyNewerVersion
)

configure_package_config_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/cmake/MyLibConfig.cmake.in"
    "${CMAKE_CURRENT_BINARY_DIR}/MyLibConfig.cmake"
    INSTALL_DESTINATION lib/cmake/MyLib
)

install(FILES
    "${CMAKE_CURRENT_BINARY_DIR}/MyLibConfig.cmake"
    "${CMAKE_CURRENT_BINARY_DIR}/MyLibConfigVersion.cmake"
    DESTINATION lib/cmake/MyLib
)
```

## 高级配置

### 条件编译

```cmake
# 根据不同平台设置不同的 Qt 路径
if(CMAKE_SYSTEM_NAME MATCHES "Linux")
    set(QT_PATH "/opt/qt-6.8.1")
elseif(CMAKE_SYSTEM_NAME MATCHES "Windows")
    set(QT_PATH "C:/Qt/6.8.1/msvc2019_64")
elseif(CMAKE_SYSTEM_NAME MATCHES "Darwin")
    set(QT_PATH "/usr/local/Qt-6.8.1")
endif()
```

### 多 Qt 版本切换

```cmake
# 允许通过命令行参数指定 Qt 路径
set(QT_PATH "" CACHE PATH "Qt installation path")

if(NOT QT_PATH)
    # 默认路径
    set(QT_PATH "/opt/qt-6.8.1")
endif()

message(STATUS "Using Qt from: ${QT_PATH}")
list(APPEND CMAKE_PREFIX_PATH "${QT_PATH}")
```

使用方式：

```bash
cmake -DQT_PATH=/opt/qt-6.9.0 ..
```

### 静态链接 Qt

如果你的 Qt 是静态编译的：

```cmake
# 静态链接 Qt
target_link_libraries(myapp PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
)

# Windows 静态链接需要额外的配置
if(WIN32)
    set_target_properties(myapp PROPERTIES
        LINK_FLAGS "/STATIC"
    )
endif()
```

## 常见问题

### Q: CMake 找不到 Qt6

**A:** 确保 `CMAKE_PREFIX_PATH` 正确设置：

```bash
cmake -DCMAKE_PREFIX_PATH=/opt/qt-6.8.1 ..
```

或检查 `/opt/qt-6.8.1/lib/cmake/Qt6/Qt6Config.cmake` 是否存在。

### Q: MOC 文件没有生成

**A:** 确保启用了 `CMAKE_AUTOMOC`：

```cmake
set(CMAKE_AUTOMOC ON)
```

### Q: 运行时找不到 Qt 库

**A:** 设置 `RPATH`：

```cmake
set_target_properties(myapp PROPERTIES
    INSTALL_RPATH "${QT_PATH}/lib"
    BUILD_WITH_INSTALL_RPATH TRUE
)
```

或设置环境变量：

```bash
export LD_LIBRARY_PATH=/opt/qt-6.8.1/lib:$LD_LIBRARY_PATH
```
