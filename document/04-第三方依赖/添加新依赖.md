# 添加新的第三方依赖

本文档介绍如何向 Qt 编译管线添加新的第三方依赖库。

## 步骤一：在 config/third_party.conf 中添加配置

首先，需要在 `config/third_party.conf` 配置文件中添加新库的配置项。

### 配置项说明

```bash
# 库的基本信息配置
LIB_NAME_VERSION="x.y.z"           # 库的版本号
LIB_NAME_URL="https://example.com/lib_name-${LIB_NAME_VERSION}.tar.gz"  # 下载地址
LIB_NAME_SHA256="..."               # 文件的 SHA256 校验和（可选，推荐）

# 构建配置
LIB_NAME_BUILD_DIR="${BUILD_DIR}/lib_name"      # 构建目录
LIB_NAME_INSTALL_DIR="${INSTALL_DIR}/lib_name"  # 安装目录
LIB_NAME_CONFIG_FILE="${BUILD_DIR}/lib_name-config.sh"  # 配置文件路径

# 依赖其他库（可选）
LIB_NAME_DEPENDS="other_lib1 other_lib2"  # 依赖的其他库名称

# 启用开关（用于命令行参数控制）
ENABLE_LIB_NAME="yes"  # 默认启用，可通过 --enable-lib-name 或 --disable-lib-name 控制
```

### 更新 THIRD_PARTY_LIBS 列表

将新库名称添加到第三方库列表中：

```bash
# 第三方库列表（按依赖顺序排列，被依赖的库在前）
THIRD_PARTY_LIBS="zlib openssl lib_name"
```

### 完整配置示例

```bash
# ============================================================
# lib_name 库配置
# ============================================================
LIB_NAME_VERSION="1.2.3"
LIB_NAME_URL="https://github.com/example/lib_name/releases/download/v${LIB_NAME_VERSION}/lib_name-${LIB_NAME_VERSION}.tar.gz"
LIB_NAME_SHA256="abcdef1234567890abcdef1234567890abcdef1234567890abcdef1234567890"

LIB_NAME_BUILD_DIR="${BUILD_DIR}/lib_name"
LIB_NAME_INSTALL_DIR="${INSTALL_DIR}/lib_name"
LIB_NAME_CONFIG_FILE="${BUILD_DIR}/lib_name-config.sh"

ENABLE_LIB_NAME="yes"

# 添加到第三方库列表（注意依赖顺序）
THIRD_PARTY_LIBS="${THIRD_PARTY_LIBS} lib_name"
```

## 步骤二：创建构建脚本目录

为新库创建专用的构建脚本目录：

```bash
scripts/third_party/lib_name/
├── builtin.sh          # 内置构建脚本（必需）
└── patches/            # 补丁目录（可选）
    └── 0001-fix-cross-compile.patch
```

创建目录：

```bash
mkdir -p scripts/third_party/lib_name/patches
```

## 步骤三：实现必需函数

在 `scripts/third_party/lib_name/builtin.sh` 中实现以下必需函数：

### 函数命名规范

所有函数名使用 `lib_name_builtin_` 前缀，其中 `lib_name` 是库名称。

### 函数模板

```bash
#!/bin/bash
# lib_name 内置构建脚本

# 源码目录
LIB_NAME_SOURCE_DIR="${LIB_NAME_BUILD_DIR}/lib_name-${LIB_NAME_VERSION}"

# ------------------------------------------------------------
# lib_name_builtin_check - 检查库是否已安装
# ------------------------------------------------------------
# 返回值：
#   0 - 已安装
#   1 - 未安装
# ------------------------------------------------------------
lib_name_builtin_check() {
    if [[ -f "${LIB_NAME_INSTALL_DIR}/lib/libname.a" ]] || \
       [[ -f "${LIB_NAME_INSTALL_DIR}/lib/libname.so" ]]; then
        log_info "lib_name 已安装"
        return 0
    fi
    return 1
}

# ------------------------------------------------------------
# lib_name_builtin_fetch - 下载源码
# ------------------------------------------------------------
lib_name_builtin_fetch() {
    log_info "下载 lib_name 源码..."

    local archive_file="${LIB_NAME_BUILD_DIR}/$(basename ${LIB_NAME_URL})"

    # 检查是否已下载
    if [[ -f "${archive_file}" ]]; then
        log_info "源码已存在，跳过下载"
        return 0
    fi

    # 创建构建目录
    mkdir -p "${LIB_NAME_BUILD_DIR}"

    # 下载源码
    if ! download_file "${LIB_NAME_URL}" "${archive_file}"; then
        log_error "下载 lib_name 源码失败"
        return 1
    fi

    # 验证 SHA256（可选）
    if [[ -n "${LIB_NAME_SHA256}" ]]; then
        if ! verify_sha256 "${archive_file}" "${LIB_NAME_SHA256}"; then
            log_error "SHA256 校验失败"
            return 1
        fi
    fi

    # 解压源码
    log_info "解压源码..."
    if ! tar -xzf "${archive_file}" -C "${LIB_NAME_BUILD_DIR}"; then
        log_error "解压失败"
        return 1
    fi

    # 应用补丁（如果有）
    local patch_dir="${SCRIPT_DIR}/third_party/lib_name/patches"
    if [[ -d "${patch_dir}" ]] && [[ -n "$(ls -A ${patch_dir}/*.patch 2>/dev/null)" ]]; then
        log_info "应用补丁..."
        for patch in "${patch_dir}"/*.patch; do
            if ! patch -d "${LIB_NAME_SOURCE_DIR}" -p1 < "${patch}"; then
                log_error "应用补丁失败: ${patch}"
                return 1
            fi
        done
    fi

    return 0
}

# ------------------------------------------------------------
# lib_name_builtin_configure - 配置交叉编译
# ------------------------------------------------------------
lib_name_builtin_configure() {
    log_info "配置 lib_name..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1

    # 根据构建系统类型配置
    # 详见下方"不同构建系统的配置示例"
    ...

    return 0
}

# ------------------------------------------------------------
# lib_name_builtin_build - 编译
# ------------------------------------------------------------
lib_name_builtin_build() {
    log_info "编译 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1

    # 根据构建系统类型编译
    # 详见下方"不同构建系统的配置示例"
    ...

    return 0
}

# ------------------------------------------------------------
# lib_name_builtin_do_install - 安装
# ------------------------------------------------------------
lib_name_builtin_do_install() {
    log_info "安装 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1

    # 根据构建系统类型安装
    # 详见下方"不同构建系统的配置示例"
    ...

    return 0
}

# ------------------------------------------------------------
# lib_name_builtin_config - 生成配置文件
# ------------------------------------------------------------
# 生成的配置文件将被 Qt 编译时引用
# ------------------------------------------------------------
lib_name_builtin_config() {
    log_info "生成 lib_name 配置文件..."

    cat > "${LIB_NAME_CONFIG_FILE}" << EOF
# lib_name 配置
export LIB_NAME_CFLAGS="-I${LIB_NAME_INSTALL_DIR}/include"
export LIB_NAME_LDFLAGS="-L${LIB_NAME_INSTALL_DIR}/lib -lname"
export LIB_NAME_LIBS="${LIB_NAME_INSTALL_DIR}/lib/libname.a"
EOF

    return 0
}

# ------------------------------------------------------------
# lib_name_builtin_install - 主安装流程
# ------------------------------------------------------------
# 参数：
#   $1 - 操作类型 (fetch|configure|build|install|clean)
# ------------------------------------------------------------
lib_name_builtin_install() {
    local action="${1:-install}"

    case "${action}" in
        fetch)
            lib_name_builtin_fetch
            ;;
        configure)
            lib_name_builtin_configure
            ;;
        build)
            lib_name_builtin_build
            ;;
        install)
            lib_name_builtin_do_install
            lib_name_builtin_config
            ;;
        clean)
            lib_name_builtin_clean
            ;;
        *)
            log_error "未知操作: ${action}"
            return 1
            ;;
    esac
}

# ------------------------------------------------------------
# lib_name_builtin_clean - 清理
# ------------------------------------------------------------
lib_name_builtin_clean() {
    log_info "清理 lib_name 构建目录..."
    rm -rf "${LIB_NAME_BUILD_DIR}"
    return 0
}
```

## 不同构建系统的配置示例

### Autotools 构建系统 (./configure)

```bash
lib_name_builtin_configure() {
    log_info "配置 lib_name (autotools)..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1

    # 配置交叉编译环境
    local configure_opts=(
        --host="${HOST_TRIPLET}"
        --prefix="${LIB_NAME_INSTALL_DIR}"
        --enable-static
        --disable-shared
        --with-pic
    )

    # 添加依赖库
    if [[ -n "${LIB_NAME_DEPENDS}" ]]; then
        for dep in ${LIB_NAME_DEPENDS}; do
            local dep_install_dir="${INSTALL_DIR}/${dep}"
            configure_opts+=(
                "CFLAGS+=-I${dep_install_dir}/include"
                "LDFLAGS+=-L${dep_install_dir}/lib"
            )
        done
    fi

    # 运行 configure
    ./configure "${configure_opts[@]}"

    return $?
}

lib_name_builtin_build() {
    log_info "编译 lib_name..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1
    make -j"${BUILD_JOBS}"

    return $?
}

lib_name_builtin_do_install() {
    log_info "安装 lib_name..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1
    make install

    return $?
}
```

### Meson 构建系统

```bash
lib_name_builtin_configure() {
    log_info "配置 lib_name (meson)..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1

    # Meson 配置选项
    local meson_opts=(
        --cross-file="${MESON_CROSS_FILE}"
        --prefix="${LIB_NAME_INSTALL_DIR}"
        --default-library=static
        -Dbuildtype=release
        -Doptimization=3
    )

    # 添加特定库的选项
    meson_opts+=(
        -Denable_tests=false
        -Denable_docs=false
    )

    meson setup "${LIB_NAME_SOURCE_DIR}" build "${meson_opts[@]}"

    return $?
}

lib_name_builtin_build() {
    log_info "编译 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1
    meson compile -C build -j "${BUILD_JOBS}"

    return $?
}

lib_name_builtin_do_install() {
    log_info "安装 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1
    meson install -C build

    return $?
}
```

### CMake 构建系统

```bash
lib_name_builtin_configure() {
    log_info "配置 lib_name (cmake)..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1

    # CMake 配置选项
    local cmake_opts=(
        -DCMAKE_TOOLCHAIN_FILE="${CMAKE_TOOLCHAIN_FILE}"
        -DCMAKE_INSTALL_PREFIX="${LIB_NAME_INSTALL_DIR}"
        -DBUILD_SHARED_LIBS=OFF
        -DCMAKE_BUILD_TYPE=Release
        -DCMAKE_POSITION_INDEPENDENT_CODE=ON
    )

    # 添加特定库的选项
    cmake_opts+=(
        -DBUILD_TESTS=OFF
        -DBUILD_EXAMPLES=OFF
    )

    cmake "${cmake_opts]}" -B build -S "${LIB_NAME_SOURCE_DIR}"

    return $?
}

lib_name_builtin_build() {
    log_info "编译 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1
    cmake --build build -j "${BUILD_JOBS}"

    return $?
}

lib_name_builtin_do_install() {
    log_info "安装 lib_name..."

    cd "${LIB_NAME_BUILD_DIR}" || return 1
    cmake --install build

    return $?
}
```

### 自定义构建系统

某些库使用自定义构建系统，需要编写特定的构建命令：

```bash
lib_name_builtin_configure() {
    log_info "配置 lib_name (自定义)..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1

    # 创建自定义配置文件
    cat > config.mk << EOF
PREFIX = ${LIB_NAME_INSTALL_DIR}
CC = ${CC}
CXX = ${CXX}
AR = ${AR}
STRIP = ${STRIP}
CFLAGS = -I${LIB_NAME_INSTALL_DIR}/include ${CFLAGS}
LDFLAGS = -L${LIB_NAME_INSTALL_DIR}/lib ${LDFLAGS}
EOF

    return 0
}

lib_name_builtin_build() {
    log_info "编译 lib_name..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1
    make -j"${BUILD_JOBS}" PREFIX="${LIB_NAME_INSTALL_DIR}"

    return $?
}

lib_name_builtin_do_install() {
    log_info "安装 lib_name..."

    cd "${LIB_NAME_SOURCE_DIR}" || return 1
    make install PREFIX="${LIB_NAME_INSTALL_DIR}"

    return $?
}
```

## 完整示例：添加一个假设的新库

以下是一个完整的示例，展示如何添加一个名为 `examplelib` 的库：

### 1. config/third_party.conf 配置

```bash
# ============================================================
# examplelib 库配置
# ============================================================
EXAMPLELIB_VERSION="2.1.0"
EXAMPLELIB_URL="https://github.com/example/examplelib/archive/refs/tags/v${EXAMPLELIB_VERSION}.tar.gz"
EXAMPLELIB_SHA256="0123456789abcdef0123456789abcdef0123456789abcdef0123456789abcdef"

EXAMPLELIB_BUILD_DIR="${BUILD_DIR}/examplelib"
EXAMPLELIB_INSTALL_DIR="${INSTALL_DIR}/examplelib"
EXAMPLELIB_CONFIG_FILE="${BUILD_DIR}/examplelib-config.sh"

# 依赖 zlib
EXAMPLELIB_DEPENDS="zlib"

ENABLE_EXAMPLELIB="yes"

# 添加到第三方库列表（放在 zlib 之后）
THIRD_PARTY_LIBS="${THIRD_PARTY_LIBS} examplelib"
```

### 2. scripts/third_party/examplelib/builtin.sh

```bash
#!/bin/bash
# examplelib 内置构建脚本

LIB_NAME="examplelib"
EXAMPLELIB_SOURCE_DIR="${EXAMPLELIB_BUILD_DIR}/${LIB_NAME}-${EXAMPLELIB_VERSION}"

# ------------------------------------------------------------
# examplelib_builtin_check - 检查库是否已安装
# ------------------------------------------------------------
examplelib_builtin_check() {
    if [[ -f "${EXAMPLELIB_INSTALL_DIR}/lib/libexample.a" ]] || \
       [[ -f "${EXAMPLELIB_INSTALL_DIR}/lib/libexample.so" ]]; then
        log_info "examplelib 已安装"
        return 0
    fi
    return 1
}

# ------------------------------------------------------------
# examplelib_builtin_fetch - 下载源码
# ------------------------------------------------------------
examplelib_builtin_fetch() {
    log_info "下载 examplelib 源码..."

    local archive_file="${EXAMPLELIB_BUILD_DIR}/$(basename ${EXAMPLELIB_URL})"

    if [[ -f "${archive_file}" ]]; then
        log_info "源码已存在，跳过下载"
        return 0
    fi

    mkdir -p "${EXAMPLELIB_BUILD_DIR}"

    if ! download_file "${EXAMPLELIB_URL}" "${archive_file}"; then
        log_error "下载 examplelib 源码失败"
        return 1
    fi

    # 验证 SHA256
    if ! verify_sha256 "${archive_file}" "${EXAMPLELIB_SHA256}"; then
        log_error "SHA256 校验失败"
        return 1
    fi

    # 解压
    log_info "解压源码..."
    if ! tar -xzf "${archive_file}" -C "${EXAMPLELIB_BUILD_DIR}"; then
        log_error "解压失败"
        return 1
    fi

    # 应用补丁（如果有）
    local patch_dir="${SCRIPT_DIR}/third_party/examplelib/patches"
    if [[ -d "${patch_dir}" ]] && [[ -n "$(ls -A ${patch_dir}/*.patch 2>/dev/null)" ]]; then
        log_info "应用补丁..."
        for patch in "${patch_dir}"/*.patch; do
            if ! patch -d "${EXAMPLELIB_SOURCE_DIR}" -p1 < "${patch}"; then
                log_error "应用补丁失败: ${patch}"
                return 1
            fi
        done
    fi

    return 0
}

# ------------------------------------------------------------
# examplelib_builtin_configure - 配置交叉编译
# ------------------------------------------------------------
examplelib_builtin_configure() {
    log_info "配置 examplelib..."

    cd "${EXAMPLELIB_SOURCE_DIR}" || return 1

    # 使用 autotools
    local configure_opts=(
        --host="${HOST_TRIPLET}"
        --prefix="${EXAMPLELIB_INSTALL_DIR}"
        --enable-static
        --disable-shared
        --with-pic
    )

    # 添加 zlib 依赖
    if [[ -d "${ZLIB_INSTALL_DIR}" ]]; then
        configure_opts+=(
            "ZLIB_CFLAGS=-I${ZLIB_INSTALL_DIR}/include"
            "ZLIB_LIBS=-L${ZLIB_INSTALL_DIR}/lib -lz"
        )
    fi

    ./configure "${configure_opts[@]}"

    return $?
}

# ------------------------------------------------------------
# examplelib_builtin_build - 编译
# ------------------------------------------------------------
examplelib_builtin_build() {
    log_info "编译 examplelib..."

    cd "${EXAMPLELIB_SOURCE_DIR}" || return 1
    make -j"${BUILD_JOBS}"

    return $?
}

# ------------------------------------------------------------
# examplelib_builtin_do_install - 安装
# ------------------------------------------------------------
examplelib_builtin_do_install() {
    log_info "安装 examplelib..."

    cd "${EXAMPLELIB_SOURCE_DIR}" || return 1
    make install

    return $?
}

# ------------------------------------------------------------
# examplelib_builtin_config - 生成配置文件
# ------------------------------------------------------------
examplelib_builtin_config() {
    log_info "生成 examplelib 配置文件..."

    cat > "${EXAMPLELIB_CONFIG_FILE}" << EOF
# examplelib 配置
export EXAMPLELIB_CFLAGS="-I${EXAMPLELIB_INSTALL_DIR}/include"
export EXAMPLELIB_LDFLAGS="-L${EXAMPLELIB_INSTALL_DIR}/lib -lexample"
export EXAMPLELIB_LIBS="${EXAMPLELIB_INSTALL_DIR}/lib/libexample.a"
EOF

    return 0
}

# ------------------------------------------------------------
# examplelib_builtin_install - 主安装流程
# ------------------------------------------------------------
examplelib_builtin_install() {
    local action="${1:-install}"

    case "${action}" in
        fetch)
            examplelib_builtin_fetch
            ;;
        configure)
            examplelib_builtin_configure
            ;;
        build)
            examplelib_builtin_build
            ;;
        install)
            examplelib_builtin_do_install
            examplelib_builtin_config
            ;;
        clean)
            examplelib_builtin_clean
            ;;
        *)
            log_error "未知操作: ${action}"
            return 1
            ;;
    esac
}

# ------------------------------------------------------------
# examplelib_builtin_clean - 清理
# ------------------------------------------------------------
examplelib_builtin_clean() {
    log_info "清理 examplelib 构建目录..."
    rm -rf "${EXAMPLELIB_BUILD_DIR}"
    return 0
}
```

## 测试新添加的库

添加完成后，可以测试新库的构建：

```bash
# 仅下载源码
./build.sh --enable-examplelib --fetch-only

# 仅配置
./build.sh --enable-examplelib --configure-only

# 仅编译
./build.sh --enable-examplelib --build-only

# 完整安装
./build.sh --enable-examplelib

# 清理
./build.sh --enable-examplelib --clean
```

## 常见问题

### 1. 库的依赖顺序

在 `THIRD_PARTY_LIBS` 列表中，被依赖的库必须放在前面。例如，如果 `libA` 依赖 `libB` 和 `libC`，则顺序应该是：

```bash
THIRD_PARTY_LIBS="libB libC libA"
```

### 2. 交叉编译问题

确保正确设置 `HOST_TRIPLET`、`CFLAGS`、`LDFLAGS` 等环境变量。某些库可能需要额外的配置选项才能正确交叉编译。

### 3. 静态库与共享库

为了便于 Qt 的静态链接，建议优先编译静态库。在配置时使用：
- Autotools: `--enable-static --disable-shared`
- Meson: `--default-library=static`
- CMake: `-DBUILD_SHARED_LIBS=OFF`

### 4. PIC (Position Independent Code)

为了确保静态库能正确链接到共享库中，需要启用 PIC：
- Autotools: `--with-pic`
- CMake: `-DCMAKE_POSITION_INDEPENDENT_CODE=ON`
- Meson: 通常默认启用
